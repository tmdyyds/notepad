#### ipv4
- 正则验证
```
function isValidIpAddress(string $ipAddress)
{
    $regex = '/^(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|[1-9])\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)$/';
    return preg_match($regex, $ipAddress);
}
```
