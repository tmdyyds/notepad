//动态规划
func fib(n int) int {
	dp := make(map[int]int, 30)
	dp[1] = 1
	dp[2] = 1
	for i := 3; i <= n; i++ {
		dp[i] = dp[i-1] + dp[i-2]
	}

	return dp[n]
}

//题解四
func fib(n int) int {
	if n < 2 {
		return n
	}

	prev := 1
	curr := 1
	for i := 3; i <= n; i++ {
		//根据斐波那契数特性交换值
		sum := prev + curr
		prev = curr
		curr = sum
	}
	return curr
}
